#Siempre Creen su propio entorno virtual, el que esta en el proyecto esta corrupto
#Para ello ingresen lo siguiente

Remove-Item -Recurse -Force .venv

#Luego creen el entorno virtual con este comando
py -m venv .venv

#Lo activan con este comando
.\.venv\Scripts\activate

#Ingresaan estos comando para instalar dependencias
python -m pip install --upgrade pip
python -m pip install fastapi uvicorn sqlalchemy psycopg2-binary python-dotenv bcrypt python-jose

#Para levantar el proyecto deben de estar dentro de backend y ejecutar el siguiente comando
uvicorn main:app --reload
Esto tendra que levanta lo siguiente
http://127.0.0.1:8000/
